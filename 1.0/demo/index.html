            Mqa.add('landscape', "(orientation:landscape)").
            //立即执行
            on('landscape', createRet, true).
            //直接通过媒体查询字符串判断最小宽度
            on("(min-width:400px)", createRet).
            //添加媒体查询字符串的假名
            add('smallscreen', "(min-width:400px)").
            //到这里，可直接对假名进行事件监听
            on('smallscreen', function (mql) {
            createRet(mql)
            })
